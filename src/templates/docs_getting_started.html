Quick Start

Get up and running with the AI-CBT API in under 5 minutes.

Step 1: Get Your API Key

First, you'll need to create an account and get your API key:

  1. Sign up for a free account
  2. Go to your dashboard
  3. Navigate to API Keys
  4. Create your first API key

Step 2: Make Your First Request

Test the API with a simple scoring request:

curl -X POST "{{ request.url_root }}api/v1/score" \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "student_answer": "Photosynthesis is the process by which plants convert light energy into chemical energy",
    "marking_guide": "Should mention light energy conversion, chlorophyll, and glucose production",
    "question_text": "Explain the process of photosynthesis"
  }'

Step 3: Handle the Response

The API will return a JSON response with the score and feedback:

{
  "success": true,
  "data": {
    "score": 4.2,
    "confidence": 0.89,
    "feedback": "Good explanation covering the basic process. Consider mentioning chlorophyll and glucose production for a complete answer.",
    "processing_time": 1.234
  },
  "meta": {
    "api_version": "v1",
    "timestamp": "2025-01-21T10:30:00Z",
    "request_id": "req_abc123"
  }
}

Code Examples

Python

import requests
import json

# Your API configuration
API_KEY = "your_api_key_here"
BASE_URL = "{{ request.url_root }}api/v1"

def score_answer(student_answer, marking_guide, question_text):
    headers = {
        "X-API-Key": API_KEY,
        "Content-Type": "application/json"
    }
    
    data = {
        "student_answer": student_answer,
        "marking_guide": marking_guide,
        "question_text": question_text
    }
    
    response = requests.post(f"{BASE_URL}/score", headers=headers, json=data)
    
    if response.status_code == 200:
        result = response.json()
        return result["data"]
    else:
        print(f"Error: {response.status_code}")
        print(response.json())
        return None

# Example usage
result = score_answer(
    student_answer="Photosynthesis converts sunlight to energy",
    marking_guide="Should explain light energy conversion process",
    question_text="How does photosynthesis work?"
)

if result:
    print(f"Score: {result['score']}/5")
    print(f"Confidence: {result['confidence']:.2%}")
    print(f"Feedback: {result['feedback']}")

JavaScript (Node.js)

const axios = require('axios');

class AICBTClient {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseURL = '{{ request.url_root }}api/v1';
    }

    async scoreAnswer(studentAnswer, markingGuide, questionText) {
        try {
            const response = await axios.post(`${this.baseURL}/score`, {
                student_answer: studentAnswer,
                marking_guide: markingGuide,
                question_text: questionText
            }, {
                headers: {
                    'X-API-Key': this.apiKey,
                    'Content-Type': 'application/json'
                }
            });

            return response.data.data;
        } catch (error) {
            console.error('API Error:', error.response?.data || error.message);
            throw error;
        }
    }
}

// Example usage
const client = new AICBTClient('your_api_key_here');

async function example() {
    try {
        const result = await client.scoreAnswer(
            "Photosynthesis converts sunlight to energy",
            "Should explain light energy conversion process",
            "How does photosynthesis work?"
        );

        console.log(`Score: ${result.score}/5`);
        console.log(`Confidence: ${(result.confidence * 100).toFixed(1)}%`);
        console.log(`Feedback: ${result.feedback}`);
    } catch (error) {
        console.error('Failed to score answer:', error);
    }
}

example();

PHP

<?php

class AICBTClient {
    private $apiKey;
    private $baseURL;

    public function __construct($apiKey) {
        $this->apiKey = $apiKey;
        $this->baseURL = '{{ request.url_root }}api/v1';
    }

    public function scoreAnswer($studentAnswer, $markingGuide, $questionText) {
        $data = json_encode([
            'student_answer' => $studentAnswer,
            'marking_guide' => $markingGuide,
            'question_text' => $questionText
        ]);

        $context = stream_context_create([
            'http' => [
                'method' => 'POST',
                'header' => [
                    'X-API-Key: ' . $this->apiKey,
                    'Content-Type: application/json',
                    'Content-Length: ' . strlen($data)
                ],
                'content' => $data
            ]
        ]);

        $response = file_get_contents($this->baseURL . '/score', false, $context);
        
        if ($response === FALSE) {
            throw new Exception('API request failed');
        }

        $result = json_decode($response, true);
        
        if ($result['success']) {
            return $result['data'];
        } else {
            throw new Exception('API error: ' . $result['error']);
        }
    }
}

// Example usage
$client = new AICBTClient('your_api_key_here');

try {
    $result = $client->scoreAnswer(
        "Photosynthesis converts sunlight to energy",
        "Should explain light energy conversion process",
        "How does photosynthesis work?"
    );

    echo "Score: " . $result['score'] . "/5\n";
    echo "Confidence: " . round($result['confidence'] * 100, 1) . "%\n";
    echo "Feedback: " . $result['feedback'] . "\n";
} catch (Exception $e) {
    echo "Error: " . $e->getMessage() . "\n";
}

?>

Error Handling

The API returns different status codes for various error conditions:

Common Error Responses
Status Code Error Type Description
400 Bad Request Missing or invalid parameters
401 Unauthorized Invalid or missing API key
429 Rate Limit Too many requests
500 Server Error Internal server error
